#include <stdio.h>
void main() {
	int num1 = 7;
	float num2 = 5.5;
	char ch = 'w';
	printf("Result1 = %d\n", (num1 > 5));
	printf("Result2 = %d\n", ((num1 + num2) <= 10));
	printf("Result3 = %d\n", (ch == 119));
	printf("Result4 = %d\n", (ch != 'p'));
	printf("Result5 = %d\n", (ch >= 10 * (num1 + num2)));
}
