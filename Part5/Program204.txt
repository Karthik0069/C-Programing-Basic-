#include<stdio.h>
int main()
{
	printf("The size of char data type = %u\n",sizeof(char));
	printf("The size of int data type = %u\n",sizeof(int));
	printf("The size of short int data type = %u\n",sizeof(short int));
	printf("The size of long int data type = %u\n",sizeof(long int));
	printf("The size of float data type = %u\n",sizeof(float));
	printf("The size of double data type = %lu\n",sizeof(double));
	printf("The size of long double data type = %lu\n",sizeof(long double));
	return 0;
}
